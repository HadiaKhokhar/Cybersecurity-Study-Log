START

SET secretPIN
SET attempts = 0
Set FOUND = false

FOR i = 0 TO 9999

    guess = format i to 4 digits
    attempts = attempts + 1

    IF guess == secretPIN THEN
        PRINT "PIN FOUND"
        PRINT guess
        PRINT attempts
        FOUND =  True
        STOP
    END IF

    IF attempts % 5 == 0 THEN
            PRINT "LOCKED"
            WAIT longer delay
    END IF

END FOR

IF NOT FOUND
    PRINT "PIN NOT FOUND"
END IF

END
